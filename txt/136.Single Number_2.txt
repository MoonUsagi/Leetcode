class Solution {
public:
    int singleNumber(vector<int>& nums) {
        
        if (nums.size()<0)
        return 0;

        unordered_set<int> hash;
            
        for(int i=0; i<nums.size();i++)
        {
            if(hash.find(nums[i])==hash.end())
                hash.insert(nums[i]);
            else
                hash.erase(nums[i]);
        }
        auto a = hash.begin();
        return *a;
    }
};
